 <script language="javascript">

	// Für Budget Ads Option

	// Für ChatGPT
	// HINWEIS: Der gesamte JavaScript Code für ChatGPT soll nach Möglichkeit in eine externe Datei gebracht werden und so, dass nicht jeder drauf zugreifen kann!!
	async function sendMessage() {
        const apiKey = 'sk-proj-SSPZfhVIj3xgPJc3s89YT3BlbkFJdszK7kntDNaPoCOlMD4R'; // API-Key
        const userInput = document.getElementById('userInput').value;
        const responseDiv = document.getElementById('response');

        if (!userInput.trim()) {
            alert('Bitte fülle zuerst die ganzen Felder oben aus und klicke dann auf Prompt generieren. Der Prompt wird dann ausgegeben und dann kannst du erst dein Prompt an ChatGPT senden.');
            return;
        }

        const payload = {
            model: 'gpt-3.5-turbo',
            messages: [{ role: 'user', content: "Sprich persönlich." + userInput }],  // Hier kommen die benutzerdefinierte Regeln hin!!!!
        };

        responseDiv.innerHTML = 'Lädt...';

        try {
            const response = await fetch('https://api.openai.com/v1/chat/completions', {
                method: 'POST',
                headers: {
                    'Content-Type': 'application/json',
                    'Authorization': `Bearer ${apiKey}`
                },
                body: JSON.stringify(payload)
            });

            const data = await response.json();

            if (response.ok) {
                responseDiv.innerHTML = `<p><strong>Antwort:</strong> ${data.choices[0].message.content}</p>`;
            } else {
                responseDiv.innerHTML = `<p><strong>Fehler:</strong> ${data.error.message}</p>`;
            }
        } catch (error) {
            responseDiv.innerHTML = `<p><strong>Fehler:</strong> ${error.message}</p>`;
        }
    }
 </script>
